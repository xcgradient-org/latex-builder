\mainhead{P\`AGINA 1 \textbullet\ BASE I CONVERSIONS}

\subhead{Constants fixes i criteri}
\[
g=9{,}81\,\mathrm{m/s^2}
\qquad
\frac{1}{2}g\approx4{,}9\,\mathrm{m/s^2}
\]
\begin{itemize}
\item Treballa amb 2 decimals al resultat final.
\item Fes els c\`alculs en SI (m, s, m/s, m/s$^2$).
\item Tria eix i signe abans de substituir valors.
\end{itemize}

\subhead{Conversi\'o d'unitats (imprescindible)}
\[
1\,\mathrm{m/s}=3{,}6\,\mathrm{km/h}
\qquad
1\,\mathrm{km/h}=\frac{1}{3{,}6}\,\mathrm{m/s}
\]
\[
\mathrm{km/h}\rightarrow\mathrm{m/s}:\div 3{,}6
\qquad
\mathrm{m/s}\rightarrow\mathrm{km/h}:\times 3{,}6
\]
\[
36\,\mathrm{km/h}\div 3{,}6=10\,\mathrm{m/s}
\qquad
10\,\mathrm{m/s}\times 3{,}6=36\,\mathrm{km/h}
\]

\subhead{MRU (moviment rectilini uniforme)}
\[
a=0
\qquad
x=x_0+vt
\qquad
\Delta x=vt
\qquad
v=\frac{\Delta x}{\Delta t}
\]
\begin{itemize}
\item La velocitat es mant\'e constant.
\item A la gr\`afica $x-t$, la pendent \`es $v$.
\end{itemize}

\subhead{MRUA (acceleraci\'o constant)}
\[
a=\frac{v-v_0}{t}
\qquad
v=v_0+at
\]
\[
x=x_0+v_0t+\frac{1}{2}at^2
\qquad
v^2=v_0^2+2a(x-x_0)
\]
\[
\bar v=\frac{v_0+v}{2}
\qquad
\Delta x=\bar v\,t
\]

\subhead{Despejos express (surten molt)}
\[
t=\frac{v-v_0}{a}
\qquad
a=\frac{v^2-v_0^2}{2\Delta x}
\]
\[
t=\frac{2\Delta x}{v_0+v}
\qquad
v=\frac{\Delta x}{t}
\]
\examnote{Si et demanen justificaci\'o: parteix del MRUA base.}

\mainhead{P\`AGINA 2 \textbullet\ CAIGUDA LLIURE I TIR VERTICAL}

\subhead{Caiguda lliure (eix positiu cap amunt)}
\[
a=-g=-9{,}81
\]
\[
y=y_0+v_0t-4{,}9t^2
\qquad
v=v_0-9{,}81t
\]
\begin{itemize}
\item Si puja: $v_0>0$.
\item Si cau des de rep\`os: $v_0=0$.
\end{itemize}

\subhead{Cas directe: caiguda des del rep\`os}
\[
h=4{,}9t^2
\qquad
t_{\text{caig}}=\sqrt{\frac{h}{4{,}9}}
\]
\[
v=-9{,}81t
\qquad
|v|=9{,}81t
\]

\subhead{Tir vertical (eix positiu cap amunt)}
\[
a=-9{,}81
\qquad
v=v_0-9{,}81t
\]
\[
y=y_0+v_0t-4{,}9t^2
\]
\[
t_{\max}=\frac{v_0}{9{,}81}
\qquad
y_{\max}=y_0+\frac{v_0^2}{19{,}62}
\]
\[
\text{si torna al mateix nivell:}\qquad t_{\text{total}}=\frac{2v_0}{9{,}81}
\]

\subhead{Guio curt per resoldre a l'examen}
\begin{itemize}
\item 1) Escriu dades i passa-les a SI.
\item 2) Defineix eix i signe (+ i -).
\item 3) Tria l'equaci\'o on surt la inc\`ognita.
\item 4) Ailla la inc\`ognita abans de substituir.
\item 5) Comprova unitats i arrodoneix a 2 decimals.
\end{itemize}

\subhead{Recordatori final}
\examnote{Aquesta xuleta dona les f\'ormules finals. El dia de l'examen has de demostrar-les des de les equacions base (per exemple, des de $y=y_0+v_0t-4{,}9t^2$).}
